LeetCode 算法面试分类解法 About-Linked-List

前言

通常来讲，链表中的问题都是来操作next指针，这个next指针就像一根针线一样连接着下一个节点。很多链表中的问题就像穿针引线一样，其实我们要确立的就是操纵节点的next指针。

206. Reverse Linked List

/**
 * Reverse a singly linked list.
 */

反转链表，很基础的一道题，却有很多公司考察。注意保存前一个和后一个节点，所以我们需要三个指针。

/**
 * Created by Thpffcj on 2017/9/21
 */

#include <iostream>

using namespace std;

struct ListNode {
    int val;
    ListNode *next;
    ListNode(int x) : val(x), next(NULL) {}
};

// 时间复杂度: O(n)
// 空间复杂度: O(1)
class Solution {
public:
    ListNode* reverseList(ListNode* head) {

        ListNode* pre = NULL;
        ListNode* current = head;
        while ( current != NULL) {
            ListNode* next = current->next;

            current->next = pre;
            pre = current;
            current = next;
        }

        return pre;
    }
};

有时候链表的问题我们需要使用多个指针保存数据信息，使用它们来穿针引线。

203. Remove Linked List Elements

/**
 * Remove all elements from a linked list of integers that have value val.
 * Example
 * Given: 1 --> 2 --> 6 --> 3 --> 4 --> 5 --> 6, val = 6
 * Return: 1 --> 2 --> 3 --> 4 --> 5
 */

删除指定节点。

/**
 * Created by Thpffcj on 2017/9/21
 */

#include <iostream>

using namespace std;

///Definition for singly-linked list.
struct ListNode {
    int val;
    ListNode *next;
    ListNode(int x) : val(x), next(NULL) {}
};

// 不使用虚拟头结点
class Solution {
public:
    ListNode* removeElements(ListNode* head, int val) {

        while( head != NULL && head->val == val ){

            ListNode* node = head;
            head = head->next;
            delete node;
        }

        if( head == NULL) {
            return NULL;
        }

        ListNode* cur = head;
        while ( cur->next != NULL) {

            if (cur->next->val == val) {
                ListNode* delNode = cur->next;
                cur->next = delNode->next;
                delete delNode;
            } else {
                cur = cur->next;
            }
        }

        return head;
    }
};

在处理和链表相关的问题时经常会使用一个技术：设立链表的虚拟头结点。

/**
 * Created by Thpffcj on 2017/9/21
 */

#include <iostream>

using namespace std;

//Definition for singly-linked list.
struct ListNode {
    int val;
    ListNode *next;
    ListNode(int x) : val(x), next(NULL) {}
};

// 使用虚拟头结点
class Solution {
public:
    ListNode* removeElements(ListNode* head, int val) {

        ListNode* dummyHead = new ListNode(0);
        dummyHead->next = head;

        ListNode* cur = dummyHead;
        while( cur->next != NULL ){

            if( cur->next->val == val ){
                ListNode* delNode = cur->next;
                cur->next = delNode->next;
                delete delNode;
            }
            else
                cur = cur->next;
        }

        ListNode* retNode = dummyHead->next;
        delete dummyHead;

        return retNode;
    }
};

24. Swap Nodes in Pairs

/**
 * Given a linked list, swap every two adjacent nodes and return its head.
 * For example,
 * Given 1->2->3->4, you should return the list as 2->1->4->3.
 * Your algorithm should use only constant space. You may not modify the values in the list, only nodes itself can be changed.
 */

之后我们要做的事情就是将P指针挪向node1，也就是下面一对节点之前的节点，我们现在根据新的p的位置，下一个节点就是node1，下一个节点是node2，在下一个节点是next。

/**
 * Created by Thpffcj on 2017/9/21
 */

#include <iostream>

using namespace std;

// Definition for singly-linked list.
struct ListNode {
    int val;
    ListNode *next;
    ListNode(int x) : val(x), next(NULL) {}
};

class Solution {
public:
    ListNode* swapPairs(ListNode* head) {

        ListNode* dummyHead = new ListNode(0);
        dummyHead->next = head;

        ListNode* p = dummyHead;
        while ( p->next && p->next->next) {
            ListNode* node1 = p->next;
            ListNode* node2 = node1->next;
            ListNode* next = node2->next;

            node2->next = node1;
            node1->next = next;
            p->next = node2;

            p = node1;
        }

        ListNode* retNode = dummyHead->next;
        delete dummyHead;

        return head;
    }
};

其实当我们分析清楚以后，每个指针到底指向谁，代表谁，在他们之间穿针引线并没有那么难。很多时候我们觉得和链表相关的问题有些绕，重要的原因也许是你没有将每一个变量它所对应的那个节点到底含义是什么明确清楚

237. Delete Node in a Linked List

/**
 * Write a function to delete a node (except the tail) in a singly linked list, given only access to that node.
 * Supposed the linked list is 1 -> 2 -> 3 -> 4 and you are given the third node with value 3, the linked list should become
 * 1 -> 2 -> 4 after calling your function.
 */

之前我们是要靠待删除的前一个节点，通过它来检测我们删除哪一个节点，然后对于前一个节点的next指针直接指向要删除的节点的下一个节点。可是对于这个题，我们不可能拿到前一个节点了。对于这种特殊的情况，我们似乎只能通过改变链表的值，将删除节点的下一个节点的值赋值给当前节点，然后删除下一节点解决问题。不过一般链表的问题是不允许通过改变链表节点的值来解决问题。

/**
 * Created by Thpffcj on 2017/9/21
 */

#include <iostream>
#include <cassert>

using namespace std;

// Definition for singly-linked list.
struct ListNode {
    int val;
    ListNode *next;
    ListNode(int x) : val(x), next(NULL) {}
};

class Solution {
public:
    void deleteNode(ListNode* node) {

//        assert(node != NULL && node->next != NULL);
        if ( node == NULL) {
            return;
        }

        if ( node->next == NULL) {
            delete node;
            node = NULL;
            return;
        }

        node->val = node->next->val;
        ListNode* delNode = node->next;
        node->next = delNode->next;

        delete delNode;
        return;
    }
};

19. Remove Nth Node From End of List

/**
 * Given a linked list, remove the nth node from the end of list and return its head.
 * For example,
 * Given linked list: 1->2->3->4->5, and n = 2.
 * After removing the second node from the end, the linked list becomes 1->2->3->5.
 * Note:
 * Given n will always be valid.
 * Try to do this in one pass.
 */

由于我们要删除倒数第n个节点，我们却不知道整个链表的长度，我们就不知道它正数的位置，就很难从头遍历的时候删除这个节点。所以说，我们首先先遍历一遍计算链表长度，在遍历一遍删除倒数第n个节点。

/**
 * Created by Thpffcj on 2017/9/21
 */

#include <iostream>
#include <cassert>

using namespace std;

//Definition for singly-linked list.
struct ListNode {
    int val;
    ListNode *next;
    ListNode(int x) : val(x), next(NULL) {}
};

// 先记录链表总长度
// 需要对链表进行两边遍历
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {

        ListNode* dummyHead = new ListNode(0);
        dummyHead->next = head;

        int length = 0;
        for(ListNode* cur = dummyHead->next ; cur != NULL ; cur = cur->next )
            length ++;
        //cout<<length<<endl;

        int k = length - n;
        assert( k >= 0 );
        ListNode* cur = dummyHead;
        for( int i = 0 ; i < k ; i ++ )
            cur = cur -> next;

        ListNode* delNode = cur->next;
        cur->next = delNode->next;
        delete delNode;

        ListNode* retNode = dummyHead->next;
        delete dummyHead;
        return retNode;
    }
};

这个思路遍历了两遍链表，我们可以不可以优化呢？

两个指针间的距离是题目给的n，当后面一个指针移动到最后时，前一个指针指向的就是要删除的节点的前一个节点。

/**
 * Created by Thpffcj on 2017/9/21
 */
#include <iostream>
#include <cassert>

using namespace std;

//Definition for singly-linked list.
struct ListNode {
    int val;
    ListNode *next;
    ListNode(int x) : val(x), next(NULL) {}
};

// 使用双指针, 对链表只遍历了一遍
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {

        assert( n>=0 );

        ListNode* dummyHead = new ListNode(0);
        dummyHead->next = head;

        ListNode* p = dummyHead;
        ListNode* q = dummyHead;
        for ( int i = 0; i < n + 1; i++) {
            assert( q );
            q = q->next;
        }

        while ( q != NULL ) {
            p = p->next;
            q = q->next;
        }

        ListNode* delNode = p->next;
        p->next = delNode->next;
        delete delNode;

        ListNode* retNode = dummyHead->next;
        delete dummyHead;
        return retNode;
    }
};
