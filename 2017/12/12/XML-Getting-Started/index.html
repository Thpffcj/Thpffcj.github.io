初识XML(未完)

名称 命名空间 备注 所含信息（不限于）
Department.xsd http://www.nju.edu.cn/schema 学校部门信息 除系/院等单位以外，学校的其他直属/行政部门等
PersonInfo.xsd http://www.nju.edu.cn/schema 个人基本信息 包含单位相关信息（Department.xsd）等
Student.xsd http://jw.nju.edu.cn/schema 学生信息 包含个人基本信息（PersonInfo.xsd）、以及其他可选的学生相关信息（包含课业成绩等）
Course.xsd http://jw.nju.edu.cn/schema 课程信息 除系/院等单位以外，学校的其他直属/行政部门等
StudentList.xsd http://jw.nju.edu.cn/schema 学生列表 引用Student.xsd 
<?xml version="1.0" encoding="UTF-8"?>
<schema xmlns="http://www.w3.org/2001/XMLSchema"
    targetNamespace="http://jw.nju.edu.cn/schema"
    xmlns:tns="http://jw.nju.edu.cn/schema"
    xmlns:nju="http://www.nju.edu.cn/schema"
        elementFormDefault="qualified">

    <import namespace="http://www.nju.edu.cn/schema"
        schemaLocation="PersonInfo.xsd" />

    <include schemaLocation="Student.xsd"/>

    <element name="学生列表" type="tns:学生列表类型"/>

    <complexType name="学生列表类型">
        <sequence minOccurs="0" maxOccurs="45">
            <element name="院系学生" type="tns:院系学生类型"/>
        </sequence>
    </complexType>

    <complexType name="院系学生类型">
        <sequence minOccurs="0" maxOccurs="4">
            <element name="学生年级" type="tns:学生年级类型"/>
        </sequence>
        <attribute name="院系编号" type="nju:部门编号类型"/>
    </complexType>

    <complexType name="学生年级类型">
        <sequence>
            <element name="学生" type="tns:学生类型"/>
        </sequence>
        <attribute name="年级" type="tns:年级类型"/>
    </complexType>

       <simpleType name="年级类型">
        <restriction base="string">
            <enumeration value="大一"/>
            <enumeration value="大二"/>
            <enumeration value="大三"/>
            <enumeration value="大四"/>
        </restriction>
    </simpleType>

    <complexType name="学生类型">
        <sequence minOccurs="0" maxOccurs="unbounded">
            <element name="学号" type="tns:学号类型"/>
            <element name="姓名" type="nju:人员名称类型"/>
        </sequence>
    </complexType>

</schema>


<?xml version="1.0" encoding="UTF-8"?>
<schema xmlns="http://www.w3.org/2001/XMLSchema"
        targetNamespace="http://jw.nju.edu.cn/schema"
        xmlns:tns="http://jw.nju.edu.cn/schema"
        xmlns:nju="http://www.nju.edu.cn/schema"
        elementFormDefault="qualified">

    <import namespace="http://www.nju.edu.cn/schema"
        schemaLocation="PersonInfo.xsd" />
    <include schemaLocation="ScoreList.xsd" />

    <element name="学生信息" type="tns:学生信息类型"/>

    <complexType name="学生信息类型">
        <sequence>
            <element name="学生基本信息" type="tns:学生基本信息类型"/>
            <element name="学生成绩" type="tns:学生成绩类型"/>
        </sequence>
    </complexType>

    <complexType name="学生基本信息类型">
        <sequence>
            <element name="学号" type="tns:学号类型"/>
            <element name="姓名" type="nju:人员名称类型"/>
            <element name="性别" type="nju:人员性别类型"/>
            <element name="出生日期" type="date"/>
            <element name="身份证号" type="nju:身份证号类型"/>
            <element name="专业" type="nju:部门名称类型"/>
            <element name="入学时间" type="date"/>
            <element name="联系方式" type="nju:手机号类型"/>
            <element name="学籍状态" type="tns:学籍状态类型"/>
        </sequence>
    </complexType>

    <simpleType name="学号类型">
        <restriction base="string">
            <length value="9"/>
        </restriction>
    </simpleType>

    <simpleType name="学籍状态类型">
        <restriction base="string">
            <enumeration value="有"/>
            <enumeration value="无"/>
        </restriction>
    </simpleType>

    <complexType name="学生成绩类型">
        <sequence minOccurs="0" maxOccurs="8">
            <element name="成绩详情" type="tns:成绩详情类型"/>
        </sequence>
    </complexType>

    <complexType name="成绩详情类型">
        <sequence>
            <element name="课程成绩列表" type="tns:课程成绩列表类型"/>
        </sequence>
        <attribute name="学期" type="tns:学期类型"/>
    </complexType>

    <simpleType name="学期类型">
        <restriction base="string">
            <enumeration value="大一第一学期"/>
            <enumeration value="大一第二学期"/>
            <enumeration value="大二第一学期"/>
            <enumeration value="大二第二学期"/>
            <enumeration value="大三第一学期"/>
            <enumeration value="大三第二学期"/>
            <enumeration value="大四第一学期"/>
            <enumeration value="大四第二学期"/>
        </restriction>
    </simpleType>

</schema>


<?xml version="1.0" encoding="UTF-8"?>
<schema xmlns="http://www.w3.org/2001/XMLSchema"
    targetNamespace="http://jw.nju.edu.cn/schema"
    xmlns:tns="http://jw.nju.edu.cn/schema"
    elementFormDefault="qualified">
    <import namespace="http://www.nju.edu.cn/schema"
        schemaLocation="PersonInfo.xsd" />

    <include schemaLocation="Student.xsd"/>
    <include schemaLocation="Course.xsd"/>

    <element name="课程成绩列表" type="tns:课程成绩列表类型"/>

    <complexType name="课程成绩列表类型">
        <sequence minOccurs="0" maxOccurs="unbounded">
            <element name="课程成绩" type="tns:课程成绩类型"/>
        </sequence>
    </complexType>

    <complexType name="课程成绩类型">
        <sequence minOccurs="0" maxOccurs="unbounded">
            <element name="成绩" type="tns:成绩类型" />
        </sequence>
        <attribute name="课程编号" type="tns:课程编号类型"/>
        <attribute name="成绩性质" type="tns:成绩性质类型"/>
    </complexType>

    <complexType name="成绩类型">
        <sequence>
            <element name="学号" type="tns:学号类型"/>
            <element name="得分" type="tns:得分类型"/>
        </sequence>
    </complexType>

    <simpleType name="成绩性质类型">
        <restriction base="string">
            <enumeration value="平时成绩"/>
            <enumeration value="作业成绩"/>
            <enumeration value="期中成绩"/>
            <enumeration value="期末成绩"/>
            <enumeration value="总评成绩"/>
        </restriction>
    </simpleType>

    <simpleType name="得分类型">
        <restriction base="int">
            <minInclusive value="0"/>
            <maxInclusive value="100"/>
        </restriction>
    </simpleType>
</schema>
